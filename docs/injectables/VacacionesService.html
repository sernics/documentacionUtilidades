import { HttpClient } from '@angular/common/http';
import { inject, Injectable } from '@angular/core';
import { EnviromentService } from '../environment/environment.service';
import { LocalStorageService } from '../localStorageService/local-storage-service.service';
import { UserInfoService } from '../user-info/user-info.service';
import { iVacacionesForm } from '@app/models/iVacacionesForm';
import { Observable } from 'rxjs';

@Injectable({
  providedIn: 'root',
})
export class VacacionesService {
  private http = inject(HttpClient);
  private environment = inject(EnviromentService);
  private localStorage = inject(LocalStorageService);
  private userInfo = inject(UserInfoService);

  /**
   * Saves the vacaciones form data by sending a POST request to the server.
   *
   * @param subject - The subject or content of the vacaciones form.
   * @param initDate - The initial date of the vacaciones period.
   * @param endDate - The end date of the vacaciones period.
   */
  saveSubject(subject: string, initDate: Date, endDate: Date) : Observable<any> {
    const headers = {
      'Content-Type': 'application/json',
      'X-Auth-Token': this.localStorage.getItem('token') as string,
    };
    return this.http
      .post(
        `${this.environment.environmentConfig.API_URL}/formularios/vacacionesForm`,
        this.generateVacacionesForm(
          subject,
          initDate,
          endDate
        ),
        { headers, responseType: 'text' }
      );
  }

  /**
   * Generates a vacaciones form object with the provided subject, initial date, and end date.
   *
   * @param subject - The subject or content of the vacaciones form.
   * @param initDate - The initial date of the vacaciones period.
   * @param endDate - The end date of the vacaciones period.
   * @returns An object representing the vacaciones form with user information and specified dates.
   */
  generateVacacionesForm(subject: string, initDate: Date, endDate: Date) {
    let userInfo = this.userInfo.getUserInfo();
    let vacacionesForm: iVacacionesForm = {
      usuario: userInfo.userName,
      numeroNomina: userInfo.nomina,
      nombre: userInfo.name,
      apellido: userInfo.sn,
      correo: userInfo.mail,
      contenido: subject,
      fechaInit: this.parseDate(initDate),
      fechaFinal: this.parseDate(endDate),
    };
    return vacacionesForm;
  }

  /**
   * Parses a given Date object and returns a string in the format day/month/year.
   *
   * @param date - The Date object to be parsed.
   * @returns A string representing the date in the format day/month/year.
   */
  parseDate(date: Date) {
    // day/month/year
    const day = date.getDate().toString().padStart(2, '0');
    const month = (date.getMonth() + 1).toString().padStart(2, '0');
    const year = date.getFullYear();
    return `${day}/${month}/${year}`;
  }
}
