import { inject, Injectable } from '@angular/core';
import { LocalStorageService } from '../localStorageService/local-storage-service.service';
import { iCommunicationForm } from '@app/models/iCommunicationForm';
import { EnviromentService } from '../environment/environment.service';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { Observable } from 'rxjs';

@Injectable({
  providedIn: 'root'
})
export class CommunicationService {
  private localStorage = inject(LocalStorageService);
  private environment = inject(EnviromentService);
  private http = inject(HttpClient);

  /**
   * Sends a communication with the specified subject, description, and files.
   *
   * @param subject - The subject of the communication.
   * @param description - The description of the communication.
   * @param files - An array of files to be sent with the communication.
   * @returns An Observable that emits the response from the server.
   *
   * This method prepares a `FormData` object containing the files and communication data,
   * sets the appropriate headers, and sends a POST request to the server.
   *
   * The files are appended to the `FormData` object with specific names based on their index:
   * - Index 0: 'fileCero'
   * - Index 1: 'fileUno'
   * - Index 2: 'fileDos'
   *
   * The communication data is appended as a JSON Blob with the name 'communicationData'.
   *
   * The request is sent to the endpoint specified by the `API_URL` in the environment configuration.
   */
  sendCommunication(
    subject: string,
    description: string,
    files: File[]
  ): Observable<any> {
    // Preparar FormData
    const formData = new FormData();

    // Generar el objeto communicationData con listFiles poblado
    const communicationData = this.generateCommunicationForm(subject, description, files);

    // Agregar archivos con nombres específicos y procesarlos uno a uno
    files.forEach((file: File, index: number) => {
      if (index === 0) {
        formData.append('fileCero', file);
        communicationData.listfiles![0] = { name: file.name, bytes: file.size };
      }
      if (index === 1) {
        formData.append('fileUno', file);
        communicationData.listfiles![1] = { name: file.name, bytes: file.size };
      }
      if (index === 2) {
        formData.append('fileDos', file);
        communicationData.listfiles![2] = { name: file.name, bytes: file.size };
      }
    });

    // Agregar communicationData como un Blob con tipo application/json
    formData.append(
      'communicationData',
      new Blob([JSON.stringify(communicationData)], { type: 'application/json' })
    );

    // Configurar headers
    const headers = new HttpHeaders({
      'X-Auth-Token': this.localStorage.getItem('token') as string,
    });

    // Log de los datos (opcional)
    for (const pair of formData.entries()) {
      console.log(`${pair[0]}, ${pair[1]}`);
    }

    // Realizar la petición
    return this.http.post(
      `${this.environment.environmentConfig.API_URL}/formularios/communicationForm`,
      formData,
      { headers: headers, responseType: 'text'}
    );
  }

  /**
   * Generates a communication form object with the provided subject, description, and files.
   *
   * @param subject - The subject of the communication.
   * @param description - The description or content of the communication.
   * @param files - Array of files to include in listFiles with name and size.
   * @returns An object implementing the ICommunicationForm interface.
   */
  private generateCommunicationForm(subject: string, description: string, files: File[]): iCommunicationForm {
    const communicationForm: iCommunicationForm = {
      asunto: subject,
      contenido: description,
      tcp: '',
      listfiles: [] // Inicializamos con 3 objetos vacíos como en tu código original
    };

    // Si no hay archivos, dejamos listFiles como está (con objetos vacíos)
    if (files.length === 0) {
      return communicationForm;
    } else {
      // El front gestiona que como maximo hayan 3 ficheros
      for (let file of files) {
        communicationForm.listfiles?.push({
          name: file.name,
          bytes: file.size,
        });
      }
    }

    return communicationForm;
  }
}
