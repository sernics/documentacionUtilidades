import { Component, inject, OnInit, Input } from '@angular/core';
import { ReactiveFormsModule, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { iEstado } from '@app/models/iEstado';
import { iSendRespuesta } from '@app/models/iSendRespuesta';
import { BackOfficeService } from '@app/services/backOfficeService/back-office.service';
import { RoleService } from '@app/services/roleService/role.service';
import { SendRespuestaService } from '@app/services/sendRespuestaService/send-respuesta.service';
import { stateAdapter } from '@app/adapters/stateAdapter';
import { Router } from '@angular/router'; // Corrección del import
import { iResponsable } from '@app/models/iResponsable';
import { ModalService } from '@app/services/modalService/modal.service';
import { FeedbackComponent } from '@app/components/utils/modals/feedback/feedback.component';

@Component({
  selector: 'app-send-respuesta',
  standalone: true,
  imports: [ReactiveFormsModule, FeedbackComponent],
  templateUrl: './send-respuesta.component.html',
  styleUrls: ['./send-respuesta.component.scss'],
})
export class SendRespuestaComponent implements OnInit {
  @Input() responsable: string = '';
  @Input() estado: iEstado | undefined;
  @Input() correoEnvio: string = '';

  protected rolesService = inject(RoleService);
  protected backOfficeService = inject(BackOfficeService);
  protected sendRespuestaService = inject(SendRespuestaService);
  protected router = inject(Router);
  protected responsables: string[] = [];
  protected estados: Map<iEstado, iEstado[]> = new Map();
  protected selectedEstado: iEstado | null = null;
  protected selectedSubestado: iEstado | null = null;
  protected responseForm: FormGroup;

  private modalService = inject(ModalService);

  constructor(private fb: FormBuilder) {
    this.responseForm = this.fb.group({
      responsable: [''],
      estado: [''],
      subestado: [''],
      respuesta: ['', [Validators.required]],
    });
  }

  ngOnInit() {
    if (this.isAdmin()) {
      this.getResponsables();
      this.getEstados();
    } else {
      this.responseForm.patchValue({
        responsable: this.responsable,
        estado: this.estado?.descripcion || '',
      });
    }
  }

  getResponsables() {
    this.sendRespuestaService.getResponsables().subscribe({
      next: (data: iResponsable[]) => {
        data.forEach((responsable) => {
          this.responsables.push(responsable.nombre);
        });
      },
      error: (err) => {
        console.error('Error al obtener responsables:', err);
      },
    });
  }

  getEstados() {
    this.backOfficeService.getEstados().subscribe({
      next: (data: iEstado[]) => {
        this.estados = stateAdapter(data);
      },
      error: (err) => {
        this.modalService.showError('Error', err.message);
      },
    });
  }

  onEstadoChange(event: Event) {
    const selectElement = event.target as HTMLSelectElement;
    const selectedCod = Number(selectElement.value);
    const estadoSeleccionado = Array.from(this.estados.keys()).find(
      (estado) => estado.pk.cod === selectedCod
    );
    this.selectedEstado = estadoSeleccionado || null;
    this.selectedSubestado = null;
    this.responseForm.patchValue({ estado: this.selectedEstado?.descripcion || '' });
    if (this.getSubestados().length > 0) {
      this.responseForm.get('subestado')?.setValidators(Validators.required);
    } else {
      this.responseForm.get('subestado')?.clearValidators();
    }
    this.responseForm.get('subestado')?.updateValueAndValidity();
  }

  onSubestadoChange(event: Event) {
    const selectElement = event.target as HTMLSelectElement;
    const selectedCodSub = Number(selectElement.value);
    const subestadoSeleccionado = this.getSubestados().find(
      (subestado) => subestado.pk.codSubEstados === selectedCodSub
    );
    this.selectedSubestado = subestadoSeleccionado || null;
    this.responseForm.patchValue({ subestado: this.selectedSubestado?.descripcion || '' });
  }

  sendRespuesta() {
    const formValues = this.responseForm.value;

    if (this.responseForm.invalid) {
      this.responseForm.markAllAsTouched();
      return;
    }

    let data: iSendRespuesta;


    if (this.isAdmin()) {
      data = {
        authority: true,
        contenido: formValues.respuesta,
        correoDestino: this.correoEnvio,
        estado: this.selectedEstado?.descripcion || '',
        idSolicitud: this.extractIdFromUrl() as unknown as number,
        nombreResponsable: formValues.responsable,
      };
    } else {
      data = {
        authority: true,
        contenido: formValues.respuesta,
        correoDestino: this.correoEnvio,
        idSolicitud: this.extractIdFromUrl() as unknown as number,
      };
    }

    this.sendRespuestaService.sendRespuesta(data).subscribe({
      next: (response) => {
        this.modalService.showSuccess(
          'Éxito',
          'La información ha sido enviada con éxito'
        );
      },
      error: (err) => {
        console.error('Error al enviar respuesta:', err);
      },
    });
  }

  isAdmin(): boolean {
    return this.rolesService.isAdmin();
  }

  selectResponsable(responsable: string) {
    if (responsable === this.responsable) {
      this.responseForm.patchValue({ responsable });
      return true;
    }
    return false;
  }

  selectEstate(state: iEstado) {
    if (this.estado?.pk.cod === state.pk.cod) {
      this.responseForm.patchValue({ estado: this.estado.pk.cod });
      this.selectedEstado = state;
      return true;
    }
    return false;
  }

  get respuestaControl() {
    return this.responseForm.get('respuesta');
  }

  get responsableControl() {
    return this.responseForm.get('responsable');
  }

  get estadoControl() {
    return this.responseForm.get('estado');
  }

  get subestadoControl() {
    return this.responseForm.get('subestado');
  }

  getSubestados(): iEstado[] {
    return this.selectedEstado && this.estados.get(this.selectedEstado)
      ? this.estados.get(this.selectedEstado)!
      : [];
  }

  private extractIdFromUrl(): string {
    return this.router.url.split('/').pop() ?? '';
  }

  getUrl(): string{
    const fullUrl = this.router.url;
    const pathIndex = fullUrl.indexOf('#/');
    if (pathIndex !== -1) {
      return fullUrl.substring(pathIndex + 2);
    }
    return fullUrl;
  }
}
