import { Component, inject, ChangeDetectionStrategy } from '@angular/core';
import { UserInfoComponent } from '../user-info/user-info.component';
import {
  FormBuilder,
  FormGroup,
  ReactiveFormsModule,
  Validators,
  FormsModule,
} from '@angular/forms';
import { VacacionesService } from '@app/services/vacacionesService/vacaciones.service';
import { debounceTime } from 'rxjs/operators';
import { ModalService } from '@app/services/modalService/modal.service';
import { FeedbackComponent } from '@app/components/utils/modals/feedback/feedback.component';

@Component({
  selector: 'app-vacaciones',
  standalone: true,
  imports: [UserInfoComponent, ReactiveFormsModule, FormsModule, FeedbackComponent],
  templateUrl: './vacaciones.component.html',
  styleUrls: ['./vacaciones.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush,
})
export class VacacionesComponent {
  protected vacacionesForm: FormGroup;
  protected minEndDate: string = '';
  protected daysDifference: number = 0;
  protected errorMessage: string | null = null;
  private vacacionesService = inject(VacacionesService);
  private fb = inject(FormBuilder);
  private modalService = inject(ModalService);

  constructor() {
    this.vacacionesForm = this.fb.group({
      startDate: [null, [Validators.required]],
      endDate: [null, [Validators.required]],
      subject: ['', [Validators.required, Validators.minLength(1)]],
    });

    this.vacacionesForm
      .get('startDate')
      ?.valueChanges.pipe(debounceTime(100))
      .subscribe(() => {
        this.updateMinEndDate();
        this.calculateDaysDifference();
      });

    this.vacacionesForm
      .get('endDate')
      ?.valueChanges.pipe(debounceTime(100))
      .subscribe(() => {
        this.calculateDaysDifference();
      });
  }

  /**
   * Updates the minimum allowed end date (`minEndDate`) based on the selected start date (`startDate`).
   * If the `endDate` is earlier than the updated `startDate`, it adjusts the `endDate` to match the `startDate`
   * without emitting a change event.
   *
   * This method ensures that the `endDate` cannot be set to a date earlier than the `startDate`.
   */
  updateMinEndDate(): void {
    const startValue = this.vacacionesForm.get('startDate')?.value;
    if (startValue) {
      this.minEndDate = startValue;
      const endValue = this.vacacionesForm.get('endDate')?.value;
      if (endValue && endValue < startValue) {
        this.vacacionesForm
          .get('endDate')
          ?.setValue(startValue, { emitEvent: false });
      }
    }
  }

  /**
   * Calculates the difference in days between the start date and end date
   * from the `vacacionesForm` form group. The result is stored in the 
   * `daysDifference` property. If the end date is earlier than the start date 
   * or if either date is missing, the difference is set to 0.
   *
   * - Adds 1 to the calculated difference to include both the start and end dates.
   * - Assumes the dates are in a valid format that can be parsed by the `Date` constructor.
   *
   * @returns {void} This method does not return a value.
   */
  calculateDaysDifference(): void {
    const startValue = this.vacacionesForm.get('startDate')?.value;
    const endValue = this.vacacionesForm.get('endDate')?.value;
    if (startValue && endValue) {
      const start = new Date(startValue);
      const end = new Date(endValue);
      if (end >= start) {
        const timeDiff = end.getTime() - start.getTime();
        this.daysDifference = Math.ceil(timeDiff / (1000 * 3600 * 24)) + 1;
      } else {
        this.daysDifference = 0;
      }
    } else {
      this.daysDifference = 0;
    }
  }

  /**
   * Handles the submission of the vacation form. Validates the form fields,
   * checks for logical errors (e.g., end date being earlier than start date),
   * and sends the data to the vacacionesService for saving. Displays success
   * or error messages based on the result of the operation.
   *
   * @remarks
   * - Ensures all form fields are touched and validated before submission.
   * - Displays an error message if the form is invalid or if the end date
   *   is earlier than the start date.
   * - Resets the form upon successful submission.
   *
   * @throws
   * - Displays an error modal with a specific or generic error message if
   *   the API call fails.
   */
  saveSubject(): void {
    this.errorMessage = null;
    this.vacacionesForm.markAllAsTouched();

    if (this.vacacionesForm.invalid) {
      this.errorMessage = 'Por favor, rellene todos los campos correctamente';
      return;
    }

    const startValue = this.vacacionesForm.get('startDate')?.value;
    const endValue = this.vacacionesForm.get('endDate')?.value;
    const subjectValue = this.vacacionesForm.get('subject')?.value.trim();

    const start = new Date(startValue);
    const end = new Date(endValue);
    if (end < start) {
      this.errorMessage =
        'La fecha de fin no puede ser anterior a la fecha de inicio';
      return;
    }

    this.vacacionesService.saveSubject(subjectValue, start, end).subscribe({
      next: (response) => {
        this.modalService.showSuccess(
          'Éxito',
          'La información ha sido enviada con éxito'
        );
        this.vacacionesForm.reset();
      },
      error: (error) => {
        let errorMessage = 'Error desconocido. Por favor, intenta de nuevo.';
        console.log(error);

        if (error && error.error && error.error.message) {
          errorMessage = error.error.message; // Si la API devuelve un campo "message"
        } else if (error && error.message) {
          errorMessage = error.message; // Si es un error genérico de HTTP
        }

        this.modalService.showError('Error', errorMessage);
        this.errorMessage = 'Error al enviar la comunicación. Intenta de nuevo.';
      },
    });
  }

  /**
   * Adjusts the height of a textarea element dynamically based on its content.
   * This method sets the height of the textarea to "auto" to reset any previously
   * set height, then adjusts it to match the scroll height of the content.
   *
   * @param event - The input event triggered by the textarea element.
   *                The target of the event is expected to be an HTMLTextAreaElement.
   */
  autoResize(event: Event): void {
    const textarea = event.target as HTMLTextAreaElement;
    textarea.style.height = 'auto';
    textarea.style.height = `${textarea.scrollHeight}px`;
  }
}
