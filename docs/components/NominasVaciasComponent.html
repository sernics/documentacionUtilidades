import { Component, Input } from '@angular/core';

@Component({
  selector: 'app-nominas-vacias',
  standalone: true,
  imports: [],
  templateUrl: './nominas-vacias.component.html',
  styleUrls: ['./nominas-vacias.component.scss'],
})
export class NominasVaciasComponent {
  @Input() fichasVacias: string[] = [];
}

<div class="container">
  <h3>Empty Payroll Entries / Nóminas Vacías</h3>
  @if (fichasVacias.length > 0) {
    <div class="payroll-list">
      @for (nomina of fichasVacias; track $index) {
        <div class="payroll-item">
          {{ nomina }}
        </div>
      }
    </div>
  } @else {
    <p>No empty payroll entries found / No se encontraron nóminas vacías.</p>
  }
</div>
