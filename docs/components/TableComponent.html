import { Component, inject, Input, effect } from '@angular/core';
import { CommonModule } from '@angular/common';
import { FormsModule } from '@angular/forms';
import { IContentType } from '@app/models/iContentType';
import { Router } from '@angular/router';
import { LocalStorageService } from '@app/services';
import { localStorageEnum } from '@app/enums/localStorageEnum';
import { iEstadoForm } from '@app/models/iEstadoForm';
import { TableService } from '@app/services/tableService/table.service';

@Component({
  selector: 'app-table',
  standalone: true,
  imports: [CommonModule, FormsModule],
  templateUrl: './table.component.html',
  styleUrl: './table.component.scss',
})
export class TableComponent {
  private tableService = inject(TableService);
  private localStorage = inject(LocalStorageService);
  private router = inject(Router);
  protected checkDescription: boolean = false;
  protected requestsFound: number = 0;
  protected currentPage: number = 0;
  private maxPage: number = 1;
  protected countComunicacion: number = 0;
  protected countFicha: number = 0;
  protected countNomina: number = 0;
  protected countVacaciones: number = 0;
  @Input() estadoInfo: iEstadoForm | null = null;

  constructor() {
    // React to signal changes
    effect(() => {
      const data = this.tableService.getDataSignal()();
      this.requestsFound = data?.cabecerasList.page.totalElements ?? 0;
      this.countComunicacion = data?.countComunicacion ?? 0;
      this.countFicha = data?.countFicha ?? 0;
      this.countNomina = data?.countNomina ?? 0;
      this.countVacaciones = data?.countVacaciones ?? 0;
      const lastPageLink = data?.cabecerasList.links.find(
        (link) => link.rel === 'last'
      )?.href;
      const pageMatch = lastPageLink?.match(/page=(\d+)/);
      const page = pageMatch ? parseInt(pageMatch[1], 10) : null;
      this.maxPage = page ? page : 1;
    });
  }

  getData(): IContentType[] | null {
    return this.tableService.getDataSignal()()?.cabecerasList.content ?? null;
  }

  goTo(id: number, tipo: string): void {
    this.localStorage.setItem(localStorageEnum.TIPO_SOLICITUD, tipo);
    this.router.navigate(['/formularios/solicitud', id]);
  }

  onPageChange(page: number): void {
    this.currentPage = page;
    this.tableService.dataTable(this.estadoInfo as iEstadoForm, page - 1);
  }

  generateIndex(): number[] {
    const indices: number[] = [];
    const range = 5;
    const start = Math.max(1, this.currentPage + 1 - Math.floor(range / 2));
    const end = Math.min(this.maxPage, start + range - 1);

    for (let i = start; i <= end; i++) {
      indices.push(i);
    }

    if (!indices.includes(this.maxPage)) {
      if (indices.length >= range) {
        indices.pop();
      }
      indices.push(this.maxPage);
    }

    return indices;
  }
}
