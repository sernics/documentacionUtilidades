import { Component, inject } from '@angular/core';
import { UserInfoComponent } from '../user-info/user-info.component';
import { CommunicationService } from '@app/services/communicationService/communication.service';
import {
  ReactiveFormsModule,
  FormBuilder,
  FormGroup,
  Validators,
} from '@angular/forms';
import { ModalService } from '@app/services/modalService/modal.service';
import { FeedbackComponent } from '../../../utils/modals/feedback/feedback.component';

@Component({
  selector: 'app-comunicacion',
  standalone: true,
  imports: [UserInfoComponent, ReactiveFormsModule, FeedbackComponent],
  templateUrl: './comunicacion.component.html',
  styleUrl: './comunicacion.component.scss',
})
export class ComunicacionComponent {
  protected fileError: string | null = null;
  protected selectedFiles: File[] = [];
  protected formError: string | null = null;
  protected communicationForm: FormGroup;
  private communicationService = inject(CommunicationService);
  private formBuilder = inject(FormBuilder);
  private modalService = inject(ModalService);

  constructor() {
    this.communicationForm = this.formBuilder.group({
      subject: ['', [Validators.required, Validators.minLength(1)]],
      description: ['', [Validators.required, Validators.minLength(1)]],
    });
  }

  /**
   * Handles the file selection event triggered by an input element.
   * Validates the selected file based on allowed extensions and the maximum number of files.
   * Updates the `selectedFiles` array if the file is valid, or sets an error message otherwise.
   *
   * @param event - The file selection event triggered by the input element.
   *
   * @remarks
   * - Allowed file extensions are `jpg` and `pdf`.
   * - A maximum of 3 files can be selected.
   * - If the file is invalid or exceeds the limit, an error message is set in `fileError`.
   *
   * @example
   * ```typescript
   * <input type="file" (change)="onFileSelected($event)" />
   * ```
   */
  onFileSelected(event: Event): void {
    const inputElement = event.target as HTMLInputElement;

    if (inputElement.files && inputElement.files.length > 0) {
      const newFile = inputElement.files[0];

      const allowedExtensions = ['jpg', 'pdf'];
      const fileExtension = newFile.name.split('.').pop()?.toLowerCase();

      if (!fileExtension || !allowedExtensions.includes(fileExtension)) {
        this.fileError = `El archivo '${newFile.name}' no es permitido. Solo se permiten archivos JPG o PDF.`;
        inputElement.value = '';
        return;
      }

      if (this.selectedFiles.length + 1 > 3) {
        this.fileError = 'Solo puedes subir hasta 3 archivos.';
        inputElement.value = '';
      } else {
        this.fileError = null;
        this.selectedFiles.push(newFile);
        inputElement.value = '';
      }
    }
  }

  /**
   * Removes a file from the selected files list at the specified index.
   * Also clears any existing file-related error.
   *
   * @param index - The index of the file to be removed from the `selectedFiles` array.
   */
  removeFile(index: number): void {
    this.selectedFiles.splice(index, 1);
    this.fileError = null;
  }

  /**
   * Sends a communication using the data provided in the form.
   * 
   * This method validates the form, ensuring all required fields are filled.
   * If the form is invalid, it displays an error message and marks all fields as touched.
   * If the form is valid, it sends the communication data (subject, description, and files)
   * to the backend service and handles the response.
   * 
   * On success:
   * - Displays a success modal with a confirmation message.
   * - Resets the form and clears the selected files.
   * 
   * On error:
   * - Logs the error to the console.
   * - Displays an error modal with a specific or generic error message.
   * - Sets a form-level error message for user feedback.
   * 
   * @returns {void}
   */
  sendCommunication(): void {
    this.formError = null;

    if (this.communicationForm.invalid) {
      this.formError = 'Por favor, rellene todos los campos obligatorios';
      this.communicationForm.markAllAsTouched();
      return;
    }

    const subject = this.communicationForm.get('subject')?.value.trim();
    const description = this.communicationForm.get('description')?.value.trim();
    const files = this.selectedFiles;

    this.communicationService
      .sendCommunication(subject, description, files)
      .subscribe({
        next: (response) => {
          this.modalService.showSuccess(
            'Éxito',
            'La información ha sido enviada con éxito'
          );
          this.communicationForm.reset();
          this.selectedFiles = [];
        },
        error: (error) => {
          let errorMessage = 'Error desconocido. Por favor, intenta de nuevo.';
          console.log(error);

          if (error && error.error && error.error.message) {
            errorMessage = error.error.message; // Si la API devuelve un campo "message"
          } else if (error && error.message) {
            errorMessage = error.message; // Si es un error genérico de HTTP
          }

          this.modalService.showError('Error', errorMessage);
          this.formError = 'Error al enviar la comunicación. Intenta de nuevo.';
        },
      });
  }
}
