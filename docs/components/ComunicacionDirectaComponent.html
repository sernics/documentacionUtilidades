import { Component, inject, Input } from '@angular/core';
import { iFile } from '@app/models/iFile';
import { iResponseCommunication } from '@app/models/responseSolicitud/iResponseCommunication';
import { tResponse } from '@app/models/responseSolicitud/tResponse';
import { DownloadFileService } from '@app/services/downloadFile/download-file.service';

@Component({
  selector: 'app-comunicacion-directa',
  imports: [],
  templateUrl: './comunicacion-directa.component.html',
  styleUrl: './comunicacion-directa.component.scss',
})
export class ComunicacionDirectaComponent {
  @Input() datos: tResponse | null = null;
  private downloadFileService = inject(DownloadFileService);

  /**
   * Retrieves the subject from the `datos` property if it exists and is not null.
   *
   * @returns {string | null} The subject (`asunto`) if available, otherwise `null`.
   */
  getSubject(): string | null {
    return this.datos !== null &&
      (this.datos as iResponseCommunication).asunto !== null
      ? (this.datos as iResponseCommunication).asunto
      : null;
  }

  /**
   * Retrieves the description from the `datos` property if it is not null.
   *
   * @returns {string} The `contenido` property of `datos` cast as `iResponseCommunication`, 
   *                   or an empty string if `datos` is null.
   */
  getDescription(): string {
    if (this.datos !== null) {
      return (this.datos as iResponseCommunication).contenido;
    }
    return '';
  }
  
  /**
   * Retrieves the receiver's name from the `datos` property if it exists and is not null.
   *
   * @returns {string | null} The name of the receiver (`nombreDestinatario`) if available, or `null` if not.
   */
  getReceivers(): string | null {
    return this.datos !== null &&
      (this.datos as iResponseCommunication).idSolicitud.nombreDestinatario !== null
      ? (this.datos as iResponseCommunication).idSolicitud.nombreDestinatario
      : null;
  }

  /**
   * Retrieves the list of file objects if available.
   *
   * @returns An array of file objects (`iFile[]`) if the `datos` property contains
   *          valid file data; otherwise, returns `null`.
   */
  getFileNames(): iFile[] | null {
    return this.datos !== null &&
      (this.datos as iResponseCommunication).files !== null &&
      (this.datos as iResponseCommunication).files.length > 0
      ? (this.datos as iResponseCommunication).files
      : null;
  }

  /**
   * Downloads a file by invoking the download service and creating a temporary link to trigger the download.
   *
   * @param file - The file object containing metadata such as the file name.
   * @remarks
   * This method uses the `downloadFileService` to fetch the file as a Blob, creates a temporary object URL,
   * and programmatically clicks a hidden anchor element to initiate the download.
   * The object URL is revoked after the download to free up resources.
   */
  downloadFile(file: iFile): void {
    this.downloadFileService.download(file).subscribe((blob: Blob) => {
      const url = window.URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = file.name;
      a.click();
      window.URL.revokeObjectURL(url);
    });
  }
}
