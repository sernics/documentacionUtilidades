import { Component, inject } from '@angular/core';
import { BackOfficeService } from '@app/services/backOfficeService/back-office.service';
import { iSolicitantesTCP } from '@app/models/iSolicitantesTCP';
import { iEstado } from '@app/models/iEstado';
import { stateAdapter } from '@app/adapters/stateAdapter';
import { FormsModule } from '@angular/forms';
import { iEstadoForm } from '@app/models/iEstadoForm';
import { TableComponent } from './table-component/table.component';
import { TableService } from '@app/services/tableService/table.service';
import { iTipoSolicitud } from '@app/models/iTipoSolicitud';

@Component({
  selector: 'app-back-office',
  imports: [FormsModule, TableComponent],
  templateUrl: './back-office.component.html',
  styleUrl: './back-office.component.scss',
})
export class BackOfficeComponent {
  private backOfficeService: BackOfficeService = inject(BackOfficeService);
  private tableService = inject(TableService);
  protected solicitantesTCP: iSolicitantesTCP;
  protected states: Map<iEstado, iEstado[]> = new Map<iEstado, iEstado[]>();
  protected tipos: iTipoSolicitud[] = [];
  protected estadoInfo: iEstadoForm;
  protected estado: string = 'Todos'; // Cambiamos el valor inicial a "Todos"
  protected subEstado: string = '';

  constructor() {
    this.estadoInfo = {
      id: null,
      responsable: '',
      solicitante: '',
      estado: 'Todos', // Ajustamos el valor inicial aquí también
      tipoSolicitud: '',
      receptor: '',
      numeroNomina: null,
      fecha1: null,
      fecha2: null,
      page: 0,
      size: 30,
      user: '',
      descripcion: null,
      asunto: '',
    };
    this.estado = 'Todos'; // Valor inicial "Todos"
    this.subEstado = '';
    this.solicitantesTCP = {
      solicitantes: [],
      responsables: [],
      tcp: [],
    };
    this.backOfficeService.getSolicitantesTCP().subscribe({
      next: (data: iSolicitantesTCP) => {
        if (data) {
          this.solicitantesTCP = data;
        }
      },
      error: (err) => {
        console.error('Error fetching solicitantes TCP:', err);
      },
    });
    this.backOfficeService.getEstados().subscribe({
      next: (data: iEstado[]) => {
        this.states = stateAdapter(data);
      },
    });
    this.searchInfo();
    this.backOfficeService.getTipos().subscribe({
      next: (data: iTipoSolicitud[]) => {
        this.tipos = data;
      },
      error: (err) => {
        console.error('Error fetching tipos:', err);
      },
    });
  }

  /**
   * Resets the form state to its initial values.
   * 
   * This method initializes the `estadoInfo` object with default values,
   * including null, empty strings, and predefined defaults for certain fields.
   * It also resets the `estado` and `subEstado` properties to their default states.
   * 
   * @private
   */
  private resetForm(): void {
    this.estadoInfo = {
      id: null,
      responsable: '',
      solicitante: '',
      estado: 'Todos', 
      tipoSolicitud: '',
      receptor: '',
      numeroNomina: null,
      fecha1: null,
      fecha2: null,
      page: 0,
      size: 30,
      user: '',
      descripcion: null,
      asunto: '',
    };
    this.estado = 'Todos'; 
    this.subEstado = '';
  }

  /**
   * Handles the form submission event.
   *
   * @param event - The event object associated with the form submission.
   * Prevents the default form submission behavior and logs a message to the console.
   */
  onSubmit(event: Event): void {
    event.preventDefault();
    console.log('Search submitted');
  }

  /**
   * Handles the reset event for the form.
   * Prevents the default form submission behavior and resets the form state.
   *
   * @param event - The event object associated with the reset action.
   */
  onReset(event: Event): void {
    event.preventDefault();
    this.resetForm();
    this.resetForm();
  }

  /**
   * Retrieves the current list of TCP applicants.
   *
   * @returns {iSolicitantesTCP} The list of TCP applicants.
   */
  getSoliciantesTCP(): iSolicitantesTCP {
    return this.solicitantesTCP;
  }

  /**
   * Retrieves an array of state descriptions.
   * 
   * The method initializes an array with a default value of "Todos" 
   * and then iterates over the `states` collection to extract and 
   * add the `descripcion` property of each state to the array twice.
   * 
   * @returns An array of state descriptions, starting with "Todos".
   */
  getStates(): string[] {
    let array: string[] = ['Todos']; // Añadimos "Todos" como primer elemento
    for (const [key] of this.states) {
      array.push(key.descripcion);
      array.push(key.descripcion);
    }
    return array;
  }

  /**
   * Retrieves the sub-states associated with the currently selected state.
   *
   * @returns An array of sub-states. If the selected state is "Todos" or if no matching state is found, 
   *          an empty array is returned.
   */
  getSubStates() {
    if (this.estado === 'Todos') {
      return []; // Si se selecciona "Todos", no hay subestados
    }
    if (this.estado === 'Todos') {
      return []; // Si se selecciona "Todos", no hay subestados
    }
    let estadoNecesario = Array.from(this.states.keys()).find(
      (estado) => estado.descripcion === this.estado
    );
    if (!estadoNecesario) {
      return [];
    }
    return this.states.get(estadoNecesario) || [];
  }

  /**
   * Retrieves the list of types.
   *
   * @returns An array containing the types.
   */
  getTipos() {
    return this.tipos;
  }

  /**
   * Determines whether a specific sub-state exists within a given state description.
   *
   * @param stateDescription - The description of the state to search for.
   * @param subStates - The description of the sub-state to check for within the specified state.
   * @returns `true` if the sub-state exists within the specified state; otherwise, `false`.
   */
  hasSubState(stateDescription: string, subStates: string): boolean {
    for (const [key, value] of this.states) {
      if (key.descripcion === stateDescription) {
        if (value.length === 0) {
          return false;
        }
        for (const subState of value) {
          if (subState.descripcion === subStates) {
            return true;
          }
        }
        return false;
      }
    }
    return false;
  }

  /**
   * Updates the `estadoInfo` object based on the current `estado` and `subEstado` values,
   * and then triggers a data table update via the `tableService`.
   *
   * The method performs the following logic:
   * - If `subEstado` is not empty and is valid for the current `estado`, it sets `estadoInfo.estado` to `subEstado`.
   * - If `estado` is 'Todos', it sets `estadoInfo.estado` to an empty string (or null, depending on backend requirements).
   * - Otherwise, it sets `estadoInfo.estado` to the value of `estado`.
   *
   * Finally, it calls the `dataTable` method of `tableService` with the updated `estadoInfo` and a default value of 0.
   */
  searchInfo() {
    if (this.subEstado !== '' && this.hasSubState(this.estado, this.subEstado)) {
      this.estadoInfo.estado = this.subEstado;
    } else if (this.estado === 'Todos') {
      this.estadoInfo.estado = ''; // O null, dependiendo de cómo tu backend maneje "sin filtro"
    } else if (this.estado === 'Todos') {
      this.estadoInfo.estado = ''; // O null, dependiendo de cómo tu backend maneje "sin filtro"
    } else {
      this.estadoInfo.estado = this.estado;
    }
    this.tableService.dataTable(this.estadoInfo, 0);
  }
}
