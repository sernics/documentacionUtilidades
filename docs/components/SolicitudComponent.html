import { Component, inject } from '@angular/core';
import { Router } from '@angular/router';
import { SolicitudService } from '@app/services/solicitudService/solicitud.service';
import { LocalStorageService } from '@app/services';
import { localStorageEnum } from '@app/enums/localStorageEnum';
import { IResponseCabecera } from '@app/models/iResponseCabecera';
import { TipoSolicitudEnum } from '@app/enums/TipoSolicitudEnum';
import { ResponseSolicitud } from '@app/clases/responseSolicitud';
import { IResponseFicha } from '@app/models/responseSolicitud/iResponseFicha';
import { tResponse } from '@app/models/responseSolicitud/tResponse';
import { iResponseCommunication } from '@app/models/responseSolicitud/iResponseCommunication';
import { iResponseNomina } from '@app/models/responseSolicitud/iResponseNomina';
import { iResponseVacaciones } from '@app/models/responseSolicitud/iResponseVacaciones';
import { Roles } from '@app/enums/roles';
import { RespuestasComponent } from './respuestas/respuestas.component';
import { FichaPlanEmergenciaComponent } from './ficha-plan-emergencia/ficha-plan-emergencia.component';
import { ComunicacionDirectaComponent } from './comunicacion-directa/comunicacion-directa.component';
import { ReclamacionNominaComponent } from './reclamacion-nomina/reclamacion-nomina.component';
import { PeticionVacacionesComponent } from './peticion-vacaciones/peticion-vacaciones.component';
import { SendRespuestaComponent } from "./send-respuesta/send-respuesta.component";

@Component({
  selector: 'app-solicitud',
  imports: [
    RespuestasComponent,
    FichaPlanEmergenciaComponent,
    ComunicacionDirectaComponent,
    ReclamacionNominaComponent,
    PeticionVacacionesComponent,
    SendRespuestaComponent
],
  templateUrl: './solicitud.component.html',
  styleUrl: './solicitud.component.scss',
})
export class SolicitudComponent {
  private solicitudService = inject(SolicitudService);
  private localStorage = inject(LocalStorageService);
  private id: string;
  protected responseHeader: IResponseCabecera | null = null;
  protected respuesta: tResponse | null = null;
  protected tipoSolicitud: string;
  protected router = inject(Router);

  constructor() {
    this.tipoSolicitud = this.localStorage.getItem(
      localStorageEnum.TIPO_SOLICITUD
    ) as string;
    this.id = this.extractIdFromUrl();
    this.solicitudService.callSolicitud(this.id).subscribe((res) => {
      this.responseHeader = res;
    });

    this.handleResponseByTipoSolicitud();
  }

  /**
   * Handles the response based on the type of request (`tipoSolicitud`).
   * This method uses a helper function to subscribe to the appropriate response
   * observable and assigns the result to the `respuesta` property.
   *
   * The method supports the following request types:
   * - `FichaPlanEmergencia`: Retrieves a response of type `IResponseFicha`.
   * - `ComunicacionDirecta`: Retrieves a response of type `iResponseCommunication`.
   * - `ReclamacionNomina`: Retrieves a response of type `iResponseNomina`.
   * - `PeticionVacaciones`: Retrieves a response of type `iResponseVacaciones`.
   *
   * @remarks
   * The `ResponseSolicitud` class is used to fetch the response based on the `id`.
   * Errors during the response retrieval are logged to the console.
   *
   */
  handleResponseByTipoSolicitud() {
    const responseClass = new ResponseSolicitud(this.id);

    const handleResponse = <T>(
      responseType: TipoSolicitudEnum,
      logMessage: string
    ) => {
      if (this.tipoSolicitud === responseType) {
        responseClass.getRespuesta<T>().subscribe({
          next: (res) => {
            this.respuesta = res as tResponse;
          },
          error: (err) => console.error(`Error al obtener ${logMessage}:`, err),
        });
      }
    };

    handleResponse<IResponseFicha>(
      TipoSolicitudEnum.FichaPlanEmergencia,
      'la ficha'
    );
    handleResponse<iResponseCommunication>(
      TipoSolicitudEnum.ComunicacionDirecta,
      'la comunicación'
    );
    handleResponse<iResponseNomina>(
      TipoSolicitudEnum.ReclamacionNomina,
      'la nómina'
    );
    handleResponse<iResponseVacaciones>(
      TipoSolicitudEnum.PeticionVacaciones,
      'las vacaciones'
    );
  }

  return() {
    this.router.navigate(['/formularios/backOffice']);
  }

  /**
   * Retrieves the ID associated with this component.
   *
   * @returns The ID as a string or number.
   */
  getId() {
    return this.id;
  }

  /**
   * Retrieves a formatted date string based on the provided type.
   *
   * @param type - The type of date to retrieve:
   *   - `0` for the creation date (`fechaCreacion`).
   *   - `1` for the closure date (`fechaCierre`).
   * @returns A formatted date string in the format `DD/MM/YYYY HH:mm` if the date exists,
   *          or `null` if the date is not available.
   *
   * The method determines the appropriate date based on the `tipoSolicitud` property
   * and the corresponding response object (`respuesta`). It formats the date if a valid
   * timestamp is found.
   *
   * The `tipoSolicitud` is expected to be one of the `TipoSolicitudEnum` values, and the
   * response object is mapped accordingly to retrieve the relevant date fields.
   */
  getFecha(type: number): string | null {
    const formatFecha = (timestamp: number | null): string | null => {
      if (!timestamp) return null;
      const date = new Date(timestamp);
      return `${date.getDate().toString().padStart(2, '0')}/${(
        date.getMonth() + 1
      )
        .toString()
        .padStart(2, '0')}/${date.getFullYear()} ${date
        .getHours()
        .toString()
        .padStart(2, '0')}:${date.getMinutes().toString().padStart(2, '0')}`;
    };

    const getTimestamp = (obj: any, type: number): number | null => {
      if (type === 1 && obj.fechaCierre == null) return null;
      return type === 0 ? obj.fechaCreacion : obj.fechaCierre;
    };

    const responseMap: Record<TipoSolicitudEnum, any> = {
      [TipoSolicitudEnum.FichaPlanEmergencia]: this.respuesta as IResponseFicha,
      [TipoSolicitudEnum.ComunicacionDirecta]: (
        this.respuesta as iResponseCommunication
      )?.idSolicitud,
      [TipoSolicitudEnum.ReclamacionNomina]: (this.respuesta as iResponseNomina)
        ?.idSolicitud,
      [TipoSolicitudEnum.PeticionVacaciones]: (
        this.respuesta as iResponseVacaciones
      )?.idSolicitud,
    };

    if (!this.respuesta) return null;
    const responseObj = responseMap[this.tipoSolicitud as TipoSolicitudEnum];

    const timestamp = getTimestamp(responseObj, type);
    return formatFecha(timestamp);
  }

  /**
   * Retrieves the name of the responsible person based on the type of request.
   *
   * @returns {string | null} The name of the responsible person if available, or `null` if not applicable.
   *
   * @remarks
   * - This method checks the `respuesta` property to determine the responsible person's name.
   * - If `respuesta` is not defined, the method returns `null`.
   * - For the `FichaPlanEmergencia` request type, it extracts the `responsable` property from the response.
   * - For other request types, it defaults to returning `null`.
   *
   * @example
   * ```typescript
   * const responsibleName = component.getResponsibleName();
   * console.log(responsibleName); // Outputs the responsible name or null
   * ```
   */
  getResponsibleName() {
    if (!this.respuesta) return null;
    switch (this.tipoSolicitud as TipoSolicitudEnum) {
      case TipoSolicitudEnum.FichaPlanEmergencia:
        return (this.respuesta as IResponseFicha).responsable;
      default:
        return null; // Provide a default return value
    }
  }

  /**
   * Determines if the current user has an admin role.
   *
   * @returns {boolean} `true` if the user's role is admin, otherwise `false`.
   */
  getRolAdmin(): boolean {
    return this.localStorage.getItem(localStorageEnum.USER_ROLE) == Roles.ADMIN;
  }

  /**
   * Generates and downloads a PDF based on the current request data.
   *
   * This method checks if the necessary data (`responseHeader`, `id`, and `tipoSolicitud`)
   * is available before attempting to generate the PDF. If any required data is missing,
   * it logs an error message to the console and exits early.
   *
   * If all required data is present, it calls the `getPdf` method of the `solicitudService`
   * to fetch the PDF. Upon a successful response, it triggers the `downloadPdf` method
   * to handle the file download. If an error occurs during the process, it logs the error
   * to the console.
   *
   * @throws Will log an error if `responseHeader` is not available.
   * @throws Will log an error if `id` or `tipoSolicitud` is missing.
   */
  getPdf() {
    if (!this.responseHeader) {
      console.error('No hay respuesta disponible para generar el PDF.');
      return;
    }

    if (!this.id || !this.tipoSolicitud) {
      console.error('Faltan datos necesarios para generar el PDF.');
      return;
    }

    this.solicitudService.getPdf(this.id, this.tipoSolicitud).subscribe({
      next: (res) => this.downloadPdf(res, this.id),
      error: (err) => console.error('Error al generar el PDF:', err),
    });
  }

  /**
   * Extracts the last segment of the current router URL, which is typically used as an identifier.
   *
   * @returns A string representing the extracted ID from the URL. If the URL does not contain any segments, an empty string is returned.
   */
  private extractIdFromUrl(): string {
    return this.router.url.split('/').pop() ?? '';
  }

  /**
   * Downloads a PDF file generated from the provided data and assigns it a filename
   * based on the given ID.
   *
   * @param data - The content of the PDF file as a BlobPart.
   * @param id - A unique identifier used to name the downloaded PDF file.
   *
   * This method creates a Blob object from the provided data, generates a temporary
   * URL for the Blob, and triggers a download of the file. After the download is
   * initiated, the temporary URL is revoked to free up resources.
   */
  private downloadPdf(data: BlobPart, id: string): void {
    const blob = new Blob([data], { type: 'application/pdf' });
    const url = window.URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = `solicitud_PDF_${id}.pdf`;
    a.click();
    window.URL.revokeObjectURL(url); // Clean up the URL object
  }
}
