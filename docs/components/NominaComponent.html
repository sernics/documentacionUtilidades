import { Component, inject } from '@angular/core';
import { UserInfoComponent } from '../user-info/user-info.component';
import { FormBuilder, FormGroup, ReactiveFormsModule, Validators } from '@angular/forms';
import { NominaService } from '@app/services/nominaService/nomina-service.service';
import { FeedbackComponent } from '@app/components/utils/modals/feedback/feedback.component';
import { ModalService } from '@app/services/modalService/modal.service';

@Component({
  selector: 'app-nomina',
  imports: [UserInfoComponent, ReactiveFormsModule, FeedbackComponent],
  templateUrl: './nomina.component.html',
  styleUrls: ['./nomina.component.scss'],
})
export class NominaComponent {
  protected subject: string = '';
  private nominaService = inject(NominaService);
  private modalService = inject(ModalService);
  errorMessage: string | null = null;
    protected nominaForm: FormGroup;
      private fb = inject(FormBuilder);

  constructor() {
      this.nominaForm = this.fb.group({
        subject: ['', [Validators.required]],
      });
    }

  /**
   * Handles the submission of the "nomina" form. Validates the form, sends the data to the service,
   * and provides feedback to the user based on the result of the operation.
   *
   * @remarks
   * - If the form is invalid, an error message is displayed prompting the user to fill in all fields.
   * - If the form is valid, the subject value is extracted and sent to the `nominaService`.
   * - On success, a success modal is displayed, and the form is reset.
   * - On error, an error modal is displayed with a specific or generic error message, and an error message is set for the user.
   *
   * @throws Will log the error details to the console if an error occurs during the service call.
   */
  saveSubject() {
    this.errorMessage = null;
    if (this.nominaForm.invalid) {
      this.errorMessage = 'Por favor, rellene todos los campos';
    } else {
      const subjectValue = this.nominaForm.get('subject')?.value;
      this.nominaService.saveSubject(subjectValue).subscribe({
        next: (response) => {
          this.modalService.showSuccess(
            'Éxito',
            'La información ha sido enviada con éxito'
          );
          this.nominaForm.reset();
        },
        error: (error) => {
          let errorMessage = 'Error desconocido. Por favor, intenta de nuevo.';
          console.log(error);

          if (error && error.error && error.error.message) {
            errorMessage = error.error.message; // Si la API devuelve un campo "message"
          } else if (error && error.message) {
            errorMessage = error.message; // Si es un error genérico de HTTP
          }

          this.modalService.showError('Error', errorMessage);
          this.errorMessage = 'Error al enviar la comunicación. Intenta de nuevo.';
        },
      });
    }
  }
}
