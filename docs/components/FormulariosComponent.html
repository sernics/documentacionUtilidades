import { Component, inject } from '@angular/core';
import { LinkerComponent } from '@app/components/linker/linker.component';
import { localStorageEnum } from '@app/enums/localStorageEnum';
import { Roles } from '@app/enums/roles';
import { LocalStorageService } from '@app/services';
import { MainMenuService } from '@app/services/mainMenuService/main-menu.service';
import { RoleService } from '@app/services/roleService/role.service';

@Component({
  selector: 'app-formularios',
  imports: [LinkerComponent],
  templateUrl: './formularios.component.html',
  styleUrl: './formularios.component.scss',
})
export class FormulariosComponent {
  private mainMenuService = inject(MainMenuService);
  private roleService = inject(RoleService);
  private localStorage = inject(LocalStorageService);

  /**
   * Retrieves the elements of the first submenu item from the main menu.
   *
   * @returns {any[]} An array of elements from the first submenu item.
   */
  getLink() {
    return this.mainMenuService.getMenu()[0].elements[0].elements;
  }

  /**
   * Checks if the current user is allowed to access a specific menu element based on their roles.
   *
   * @param $index - The index of the menu element to check.
   * @returns A boolean indicating whether the user is allowed to access the menu element.
   */
  isAllowed($index: number) {
    // I dont know why if i make this call here work but in the isAllowed function not
    this.roleService.setRoles(
      this.localStorage.getItem(localStorageEnum.USER_ROLE) as Roles[]
    );
    let checkRoles =
      this.mainMenuService.getMenu()[0].elements[0].elements[$index].roles;
    return this.roleService.isAllowed(checkRoles);
  }
}
