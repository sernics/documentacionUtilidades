import { Component } from '@angular/core';
import { Router } from '@angular/router';
import { AuthService } from '@app/services';
import { inject } from '@angular/core';
import { LinkerComponent } from '../linker/linker.component';
import { MainMenuService } from '@app/services/mainMenuService/main-menu.service';

@Component({
  selector: 'app-home',
  standalone: true,
  imports: [LinkerComponent],
  templateUrl: './home.component.html',
  styleUrl: './home.component.scss',
})
export class HomeComponent {
  username: string | null = '';
  private authService = inject(AuthService);
  private router = inject(Router);
  private mainMenuService = inject(MainMenuService);

  constructor() {
    this.username = this.authService.getUser() ?? null;
  }

  /**
   * Retrieves the username of the currently authenticated user.
   * This method calls the `getUser` method from the `authService`.
   */
  getUsername() {
    this.authService.getUser();
  }

  /**
   * Logs out the current user by calling the `logout` method of the `authService`
   * and then navigates to the login page.
   */
  logout() {
    this.authService.logout();
    this.router.navigate(['/login']);
  }

  /**
   * Retrieves the elements of the current menu item based on the current router URL.
   *
   * @returns The elements of the current menu item, or undefined if no matching menu item is found.
   */
  getLink() {
    // obtener el link actual
    return this.mainMenuService.getMenuItemByLink(this.router.url)?.elements;
  }
}
