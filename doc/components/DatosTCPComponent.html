import { Component, inject } from '@angular/core';
import { UserInfoComponent } from '../user-info/user-info.component';
import { FormsModule, ReactiveFormsModule } from '@angular/forms';
import { DatosTcpService } from '@app/services/datosTcpService/datos-tcp.service';
import { RoleService } from '@app/services/roleService/role.service';
import { iNomina } from '@app/models/iNomina';
import { iFichaPersonal } from '@app/models/iFichaPersonal';
import { FichaInfoComponent } from './ficha-info/ficha-info.component';
import { MatAutocompleteModule } from '@angular/material/autocomplete';
import { MatInputModule } from '@angular/material/input';
import { MatFormFieldModule } from '@angular/material/form-field';
import { AsyncPipe } from '@angular/common';
import { Observable } from 'rxjs';
import { map, startWith } from 'rxjs/operators';
import { FormControl } from '@angular/forms';
import { NominasVaciasComponent } from './nominas-vacias/nominas-vacias.component';
import { FichaUserInfoComponent } from './ficha-user-info/ficha-user-info.component';
import { iUser } from '@app/models/iUser';
import { Roles } from '@app/enums/roles';

@Component({
  selector: 'app-datos-tcp',
  standalone: true,
  imports: [
    FichaInfoComponent,
    ReactiveFormsModule,
    MatAutocompleteModule,
    MatInputModule,
    MatFormFieldModule,
    AsyncPipe,
    NominasVaciasComponent,
    FichaUserInfoComponent
  ],
  templateUrl: './datos-tcp.component.html',
  styleUrls: ['./datos-tcp.component.scss'],
})
export class DatosTCPComponent {
  private datosTcpService = inject(DatosTcpService);
  private roleService = inject(RoleService);
  protected searchQuery = new FormControl('');
  protected selectedEmployee: string = '';
  protected showContent: boolean = false;
  protected showTableEmpty: boolean = false;
  protected isLoadingEmpty: boolean = false;
  protected nominas: iNomina[] = [];
  protected fichas: iFichaPersonal[] = [];
  protected fichasVacias : string[]= [];
  protected filteredOptions!: Observable<iNomina[]>;

  constructor() {
    this.datosTcpService.getNominas().subscribe((nominas) => {
      this.nominas = nominas;
      this.setupAutocomplete();
    });
  }

  /**
   * Initializes the autocomplete functionality by setting up a stream of filtered options
   * based on the user's input in the `searchQuery` form control. The filtered options
   * are updated dynamically as the user types.
   *
   * @remarks
   * This method uses RxJS operators to listen for value changes in the `searchQuery` form control,
   * starting with an empty string, and applies a filtering function to generate the list of
   * suggestions.
   *
   * @example
   * // Assuming `searchQuery` is a FormControl, this will filter options as the user types:
   * this.setupAutocomplete();
   */
  setupAutocomplete() {
    this.filteredOptions = this.searchQuery.valueChanges.pipe(
      startWith(''),
      map(value => this._filter(value || ''))
    );
  }

  /**
   * Filters the list of `iNomina` objects based on the provided string value.
   * The filter is case-insensitive and checks if the `numeroNomina` or `nombreCompleto`
   * properties of each `iNomina` object include the given value.
   *
   * @param value - The string value to filter the `iNomina` objects by.
   * @returns An array of `iNomina` objects that match the filter criteria.
   */
  private _filter(value: string): iNomina[] {
    const filterValue = value.toLowerCase();
    return this.nominas.filter(item =>
      item.numeroNomina.toLowerCase().includes(filterValue) ||
      item.nombreCompleto.toLowerCase().includes(filterValue)
    );
  }

  /**
   * Handles the selection of an option from a dropdown or autocomplete component.
   * Updates the `selectedEmployee` property with the value of the selected option.
   *
   * @param event - The event object containing the selected option's details.
   */
  onOptionSelected(event: any) {
    this.selectedEmployee = event.option.value;
  }

  /**
   * Searches for an employee's record based on the selected employee.
   * 
   * If an employee is selected, it retrieves the employee's record using the `datosTcpService`
   * and updates the component's state to display the retrieved data.
   * If no employee is selected, it alerts the user to select an employee first.
   * 
   * @remarks
   * - The method relies on `this.selectedEmployee` to determine the employee to search for.
   * - The retrieved records are stored in `this.fichas`.
   * - The visibility of content and empty table messages is controlled by `this.showContent` and `this.showTableEmpty`.
   * 
   * @throws Will display an alert if no employee is selected.
   */
  searchEmployee() {
    if (this.selectedEmployee) {
      this.datosTcpService
        .getFichaEmpleado(this.selectedEmployee)
        .subscribe((fichas) => {
          this.fichas = fichas;
          this.showContent = true;
          this.showTableEmpty = false;
        });
    } else {
      alert('Por favor, selecciona un empleado antes de buscar.');
    }
  }

  /**
   * Initiates a search for empty payroll records by calling the `getEmptyNominas` service method.
   * 
   * - Displays a loading indicator while the search is in progress.
   * - Hides the table and content until the search is complete.
   * - Updates the component state with the retrieved data or logs an error if the request fails.
   * 
   * @remarks
   * This method handles both the success and error scenarios:
   * - On success: Updates the `fichasVacias` property with the retrieved data and displays the table.
   * - On error: Logs the error to the console and ensures the loading indicator is turned off.
   */
  searchEmpty() {
    this.isLoadingEmpty = true; // Activamos el loading
    this.showTableEmpty = false;
    this.showContent = false;
    
    this.datosTcpService.getEmptyNominas().subscribe({
      next: (fichasVacias) => {
        this.fichasVacias = fichasVacias;
        this.showTableEmpty = true;
        this.isLoadingEmpty = false; // Desactivamos el loading
      },
      error: (error) => {
        console.error('Error al cargar nóminas vacías:', error);
        this.isLoadingEmpty = false; // Desactivamos el loading incluso si hay error
      }
    });
  }

  /**
   * Determines if the current user has administrative privileges.
   *
   * @returns {boolean} `true` if the user is an admin, otherwise `false`.
   */
  isAdmin(): boolean {
    return this.roleService.isAdmin();
  }

  /**
   * Retrieves the list of personal records (fichas).
   *
   * @returns {iFichaPersonal[]} An array of personal records.
   */
  getFichas(): iFichaPersonal[] {
    return this.fichas;
  }

  /**
   * Retrieves user information by mapping data from the first element of the `fichas` array.
   *
   * @returns {iUser} An object containing user information, including authorities, full name,
   * email, employee number, username, and other details.
   *
   * @remarks
   * - The method assumes that the `fichas` array contains at least one element.
   * - Default values are used for some fields, such as `authorities`, `company`, `name`, and `sn`.
   * - The `authorities` field is set with a default role of `Roles.USER`.
   */
  getUserInfo(): iUser {
    // Tomamos el primer elemento de fichas
    const ficha = this.fichas[0];
    const idSolicitud = ficha.idSolicitud;

    // Mapeamos los datos de ficha y idSolicitud a iUser
    const user: iUser = {
      authorities: {
        authority: [Roles.USER], // Valor por defecto, ya que no tenemos roles en iFichaPersonal
      },
      cn: idSolicitud.nombreCompleto, // Nombre completo
      company: "", // Usamos cia o company si existe
      mail: idSolicitud.correo, // Correo
      name: "", // Primer nombre
      nomina: idSolicitud.numeroNomina, // Número de nómina
      sn:"", // Apellidos
      userName: idSolicitud.usuario, // Nombre de usuario
    };

    // Devolvemos un array con un solo elemento
    return user;
  }

  /**
   * Retrieves an array of empty "fichas" (records).
   *
   * @returns An array of strings representing the empty fichas.
   */
  getFichasVacias(): string[]{
    return this.fichasVacias
  }

  /**
   * Formats and returns a string representation of a `iNomina` object.
   *
   * @param nomina - The `iNomina` object containing employee details.
   * @returns A formatted string in the format "nombreCompleto (numeroNomina)" 
   *          if `nomina` is provided, otherwise an empty string.
   */
  displayFn(nomina: iNomina): string {
    return nomina ? `${nomina.nombreCompleto} (${nomina.numeroNomina})` : '';
  }
}
