Loads and maps data from the ficha array into the fichaForm form group.

This method searches for specific entries in the ficha array based on the tipoGrado property and assigns their values to corresponding fields in the form. It also sets a flag indicating whether a relative is associated with the company and toggles related validators.

The mapping is as follows:

  • contact1: Entry with tipoGrado equal to '1'.
  • contact2: Entry with tipoGrado equal to '2'.
  • contact3: Entry with tipoGrado equal to '3'.

If no matching entry is found for a specific tipoGrado, an empty object is used as a fallback.

  • The method uses optional chaining and default values to ensure robustness against missing data.
  • The hasRelativeInBinter flag is set based on the presence of company or puesto in contact3.
Example :
// Example usage:
this.loadFichaData();
Returns : void

import { Component, Input, inject, signal, WritableSignal } from '@angular/core';
import { ReactiveFormsModule, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { iFichaPersonal } from '@app/models/iFichaPersonal';
import { iFichaForms } from '@app/models/iFichaForms';
import { FichaService } from '@app/services/fichaService/ficha.service';
import { log } from 'node:console';
import { ModalService } from '../../../../../services/modalService/modal.service';
import { FeedbackComponent } from "../../../../utils/modals/feedback/feedback.component";
import { Router } from '@angular/router';

@Component({
  selector: 'app-ficha-info',
  standalone: true,
  imports: [ReactiveFormsModule, FeedbackComponent],
  templateUrl: './ficha-info.component.html',
  styleUrls: ['./ficha-info.component.scss'],
})
export class FichaInfoComponent {
  @Input() ficha: iFichaPersonal[] = [];
  protected FichaService = inject(FichaService);
  protected modalService = inject(ModalService);
    private router = inject(Router);
  protected hasRelativeInBinter: WritableSignal<boolean> = signal(false);
  errorMessage: string | null = null;
  fichaForm: FormGroup;

  constructor(private fb: FormBuilder) {
    this.fichaForm = this.fb.group({
      // Primer contacto
      nombre1: ['', Validators.required],
      apellidos1: ['', Validators.required],
      parentesco1: ['', Validators.required],
      direccion1: ['', Validators.required],
      phone11: ['', [Validators.required, Validators.pattern('^[0-9]+$')]],
      phone12: ['', [Validators.required, Validators.pattern('^[0-9]+$')]],

      // Segundo contacto
      nombre2: ['', Validators.required],
      apellidos2: ['', Validators.required],
      parentesco2: ['', Validators.required],
      direccion2: ['', Validators.required],
      phone21: ['', [Validators.required, Validators.pattern('^[0-9]+$')]],
      phone22: ['', [Validators.required, Validators.pattern('^[0-9]+$')]],

      // Tercer contacto (extra)
      nombreExtra: [''],
      apellidosExtra: [''],
      parentescoExtra: [''],
      direccionExtra: [''],
      company: [''],
      puesto: [''],
      phoneExtra1: [''],
      phoneExtra2: [''],
    });
  }

  /**
   * Lifecycle hook that is called when any data-bound property of a directive changes.
   * This method is triggered whenever Angular detects changes to the input properties
   * of the component. It is used here to load the ficha data whenever such changes occur.
   *
   * @remarks
   * Ensure that the `loadFichaData` method handles any necessary cleanup or reinitialization
   * to properly reflect the updated state of the component.
   */
  ngOnChanges() {
    this.loadFichaData();
  }

  /**
   * Loads and maps data from the `ficha` array into the `fichaForm` form group.
   * 
   * This method searches for specific entries in the `ficha` array based on the `tipoGrado` property
   * and assigns their values to corresponding fields in the form. It also sets a flag indicating
   * whether a relative is associated with the company and toggles related validators.
   * 
   * The mapping is as follows:
   * - `contact1`: Entry with `tipoGrado` equal to '1'.
   * - `contact2`: Entry with `tipoGrado` equal to '2'.
   * - `contact3`: Entry with `tipoGrado` equal to '3'.
   * 
   * If no matching entry is found for a specific `tipoGrado`, an empty object is used as a fallback.
   * 
   * @remarks
   * - The method uses optional chaining and default values to ensure robustness against missing data.
   * - The `hasRelativeInBinter` flag is set based on the presence of `company` or `puesto` in `contact3`.
   * 
   * @example
   * // Example usage:
   * this.loadFichaData();
   */
  loadFichaData() {
    const contact1 = this.ficha.find(f => f.tipoGrado === '1') || {} as iFichaPersonal;
    const contact2 = this.ficha.find(f => f.tipoGrado === '2') || {} as iFichaPersonal;
    const contact3 = this.ficha.find(f => f.tipoGrado === '3') || {} as iFichaPersonal;
    console.log(contact3);

    this.fichaForm.patchValue({
      nombre1: contact1.nombre || '',
      apellidos1: contact1.apellidos || '',
      parentesco1: contact1.parentesco || '',
      direccion1: contact1.direccion || '',
      phone11: contact1.telefono1 || '',
      phone12: contact1.telefono2 || '',
      nombre2: contact2.nombre || '',
      apellidos2: contact2.apellidos || '',
      parentesco2: contact2.parentesco || '',
      direccion2: contact2.direccion || '',
      phone21: contact2.telefono1 || '',
      phone22: contact2.telefono2 || '',
      nombreExtra: contact3.nombre || '',
      apellidosExtra: contact3.apellidos || '',
      parentescoExtra: contact3.parentesco || '',
      direccionExtra: contact3.direccion || '',
      company: contact3.company || '',
      puesto: contact3.puesto || '',
      phoneExtra1: contact3.telefono1 || '',
      phoneExtra2: contact3.telefono2 || '',
    });

    this.hasRelativeInBinter.set(!!contact3?.company || !!contact3?.puesto);
    this.toggleRelativeValidators();
  }

  /**
   * Toggles the state of the `hasRelativeInBinter` property and updates the related validators.
   * 
   * This method inverts the current value of `hasRelativeInBinter` and then calls
   * `toggleRelativeValidators` to ensure the form validators are updated accordingly.
   */
  toggleRelativeInBinter() {
    this.hasRelativeInBinter.set(!this.hasRelativeInBinter());
    this.toggleRelativeValidators();
  }

  /**
   * Toggles the validators for a set of form controls in the `fichaForm` based on the presence of a relative in Binter.
   * 
   * - If `hasRelativeInBinter()` returns `true`, required validators are applied to the specified form controls.
   * - If `hasRelativeInBinter()` returns `false`, all validators are cleared from the specified form controls.
   * 
   * After updating the validators, the method ensures that the validity state of each control is recalculated by calling `updateValueAndValidity()`.
   * 
   * Affected form controls:
   * - `nombreExtra`
   * - `apellidosExtra`
   * - `parentescoExtra`
   * - `direccionExtra`
   * - `company`
   * - `puesto`
   * - `phoneExtra1` (also validates numeric pattern)
   * - `phoneExtra2` (also validates numeric pattern)
   */
  toggleRelativeValidators() {
    if (this.hasRelativeInBinter()) {
      this.fichaForm.get('nombreExtra')?.setValidators(Validators.required);
      this.fichaForm.get('apellidosExtra')?.setValidators(Validators.required);
      this.fichaForm.get('parentescoExtra')?.setValidators(Validators.required);
      this.fichaForm.get('direccionExtra')?.setValidators(Validators.required);
      this.fichaForm.get('company')?.setValidators(Validators.required);
      this.fichaForm.get('puesto')?.setValidators(Validators.required);
      this.fichaForm.get('phoneExtra1')?.setValidators([Validators.required, Validators.pattern('^[0-9]+$')]);
      this.fichaForm.get('phoneExtra2')?.setValidators([Validators.required, Validators.pattern('^[0-9]+$')]);
    } else {
      this.fichaForm.get('nombreExtra')?.clearValidators();
      this.fichaForm.get('apellidosExtra')?.clearValidators();
      this.fichaForm.get('parentescoExtra')?.clearValidators();
      this.fichaForm.get('direccionExtra')?.clearValidators();
      this.fichaForm.get('company')?.clearValidators();
      this.fichaForm.get('puesto')?.clearValidators();
      this.fichaForm.get('phoneExtra1')?.clearValidators();
      this.fichaForm.get('phoneExtra2')?.clearValidators();

    }
    this.fichaForm.get('nombreExtra')?.updateValueAndValidity();
    this.fichaForm.get('apellidosExtra')?.updateValueAndValidity();
    this.fichaForm.get('parentescoExtra')?.updateValueAndValidity();
    this.fichaForm.get('direccionExtra')?.updateValueAndValidity();
    this.fichaForm.get('company')?.updateValueAndValidity();
    this.fichaForm.get('puesto')?.updateValueAndValidity();
    this.fichaForm.get('phoneExtra1')?.updateValueAndValidity();
    this.fichaForm.get('phoneExtra2')?.updateValueAndValidity();
  }

  /**
   * Handles the save operation for the ficha form.
   * 
   * This method validates the form, displays an error message if the form is invalid,
   * and sends the form data to the server if valid. Upon successful submission, a success
   * modal is displayed. If an error occurs during submission, an error modal is shown
   * with the appropriate error message.
   * 
   * @remarks
   * - The form is marked as touched if invalid to highlight validation errors.
   * - Error messages are derived from the API response or fallback to a generic message.
   * 
   * @throws Will display a modal with an error message if the API call fails.
   */
  saveFicha() {
    this.errorMessage = null;
    if (this.fichaForm.invalid) {
      this.errorMessage = 'Por favor, rellene todos los campos obligatorios correctamente';
      this.fichaForm.markAllAsTouched();
      return;
    }

    const fichaFormsData: iFichaForms = this.fichaForm.value;

    this.FichaService.saveSubject(fichaFormsData).subscribe({
      next: (response) => {
        this.modalService.showSuccess(
          'Éxito',
          'La información ha sido enviada con éxito'
        );
      },
      error: (error) => {
        let errorMessage = 'Error desconocido. Por favor, intenta de nuevo.';
        console.log(error);

        if (error && error.error && error.error.message) {
          errorMessage = error.error.message; // Si la API devuelve un campo "message"
        } else if (error && error.message) {
          errorMessage = error.message; // Si es un error genérico de HTTP
        }

        this.modalService.showError('Error', errorMessage);
        this.errorMessage = 'Error al enviar la comunicación. Intenta de nuevo.';
      },
    });
  }

  /**
   * Retrieves the current URL path from the router, excluding any hash fragment (`#`).
   * If a hash fragment is present, it extracts and returns the portion of the URL after `#/`.
   * Otherwise, it returns the full URL as is.
   *
   * @returns The extracted URL path or the full URL if no hash fragment is found.
   */
  getUrl(): string{
    const fullUrl = this.router.url;
    const pathIndex = fullUrl.indexOf('#/');
    if (pathIndex !== -1) {
      return fullUrl.substring(pathIndex + 2);
    }
    return fullUrl;
  }
}
