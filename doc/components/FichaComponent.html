import { Component, inject, signal, WritableSignal } from '@angular/core';
import { UserInfoComponent } from '../user-info/user-info.component';
import {
  ReactiveFormsModule,
  FormBuilder,
  FormGroup,
  Validators,
} from '@angular/forms';
import { iFichaForms } from '@app/models/iFichaForms';
import { FichaService } from '../../../../services/fichaService/ficha.service';
import { ModalService } from '@app/services/modalService/modal.service';
import { FeedbackComponent } from '@app/components/utils/modals/feedback/feedback.component';
@Component({
  selector: 'app-ficha',
  standalone: true,
  imports: [UserInfoComponent, ReactiveFormsModule, FeedbackComponent],
  templateUrl: './ficha.component.html',
  styleUrls: ['./ficha.component.scss'],
})
export class FichaComponent {
  protected FichaService = inject(FichaService);
  protected modalService = inject(ModalService);
  protected hasRelativeInBinter: WritableSignal<boolean> = signal(false);
  errorMessage: string | null = null;
  fichaForm: FormGroup;
  iFichaForm: iFichaForms | null = null;

  constructor(private fb: FormBuilder) {
    this.fichaForm = this.fb.group({
      nombre1: ['', Validators.required],
      apellidos1: ['', Validators.required],
      parentesco1: ['', Validators.required],
      dirreccion1: ['', Validators.required],
      phone11: ['', [Validators.required, Validators.pattern('^[0-9]+$')]],
      phone12: ['', [Validators.required, Validators.pattern('^[0-9]+$')]],

      nombre2: ['', Validators.required],
      apellidos2: ['', Validators.required],
      parentesco2: ['', Validators.required],
      direccion2: ['', Validators.required],
      phone21: ['', [Validators.required, Validators.pattern('^[0-9]+$')]],
      phone22: ['', [Validators.required, Validators.pattern('^[0-9]+$')]],

      nombreExtra: [''],
      apellidosExtra: [''],
      parentescoExtra: [''],
      direccionExtra: [''],
      company: [''],
      puesto: [''],
      phoneExtra1: [''],
      phoneExtra2: [''],
    });
  }

  /**
   * Toggles the state of the `hasRelativeInBinter` property and updates the form validators accordingly.
   * 
   * - When `hasRelativeInBinter` is set to `true`, it applies the required validators to specific form controls,
   *   ensuring that the user provides necessary information.
   * - When `hasRelativeInBinter` is set to `false`, it clears the validators from those form controls.
   * 
   * After updating the validators, it calls `updateValueAndValidity` on each affected form control to ensure
   * the form's validation state is recalculated.
   * 
   * Affected form controls:
   * - `nombreExtra`
   * - `apellidosExtra`
   * - `parentescoExtra`
   * - `direccionExtra`
   * - `company`
   * - `puesto`
   * - `phoneExtra1` (also validates numeric pattern)
   * - `phoneExtra2` (also validates numeric pattern)
   */
  toggleRelativeInBinter() {
    this.hasRelativeInBinter.set(!this.hasRelativeInBinter());

    if (this.hasRelativeInBinter()) {
      this.fichaForm.get('nombreExtra')?.setValidators(Validators.required);
      this.fichaForm.get('apellidosExtra')?.setValidators(Validators.required);
      this.fichaForm.get('parentescoExtra')?.setValidators(Validators.required);
      this.fichaForm.get('direccionExtra')?.setValidators(Validators.required);
      this.fichaForm.get('company')?.setValidators(Validators.required);
      this.fichaForm.get('puesto')?.setValidators(Validators.required);
      this.fichaForm
        .get('phoneExtra1')
        ?.setValidators([Validators.required, Validators.pattern('^[0-9]+$')]);
      this.fichaForm
        .get('phoneExtra2')
        ?.setValidators([Validators.required, Validators.pattern('^[0-9]+$')]);
    } else {
      this.fichaForm.get('nombreExtra')?.clearValidators();
      this.fichaForm.get('apellidosExtra')?.clearValidators();
      this.fichaForm.get('parentescoExtra')?.clearValidators();
      this.fichaForm.get('direccionExtra')?.clearValidators();
      this.fichaForm.get('company')?.clearValidators();
      this.fichaForm.get('puesto')?.clearValidators();
      this.fichaForm.get('phoneExtra1')?.clearValidators();
      this.fichaForm.get('phoneExtra2')?.clearValidators();
    }

    this.fichaForm.get('nombreExtra')?.updateValueAndValidity();
    this.fichaForm.get('apellidosExtra')?.updateValueAndValidity();
    this.fichaForm.get('parentescoExtra')?.updateValueAndValidity();
    this.fichaForm.get('direccionExtra')?.updateValueAndValidity();
    this.fichaForm.get('company')?.updateValueAndValidity();
    this.fichaForm.get('puesto')?.updateValueAndValidity();
    this.fichaForm.get('phoneExtra1')?.updateValueAndValidity();
    this.fichaForm.get('phoneExtra2')?.updateValueAndValidity();
  }

  /**
   * Handles the submission of the form data. Validates the form, displays error messages
   * if the form is invalid, and sends the form data to the server if valid.
   * 
   * - If the form is invalid:
   *   - Sets an error message prompting the user to fill out all required fields correctly.
   *   - Marks all form fields as touched to display validation errors.
   * 
   * - If the form is valid:
   *   - Converts the form data to the `iFichaForms` interface.
   *   - Sends the data to the server using the `FichaService.saveSubject` method.
   *   - On success:
   *     - Displays a success message using the `modalService`.
   *     - Resets the form.
   *   - On error:
   *     - Logs the error to the console.
   *     - Displays an error message using the `modalService`.
   *     - Sets a generic error message for the user.
   */
  saveSubject() {
    this.errorMessage = null;

    if (this.fichaForm.invalid) {
      this.errorMessage =
        'Por favor, rellene todos los campos obligatorios correctamente';
      this.fichaForm.markAllAsTouched();
      return;
    }

    this.iFichaForm = this.fichaForm.value as iFichaForms;
    console.log('Datos del formulario:', this.iFichaForm);

    this.FichaService.saveSubject(this.iFichaForm).subscribe({
      next: (response) => {
        this.modalService.showSuccess(
          'Éxito',
          'La información ha sido enviada con éxito'
        );
        this.fichaForm.reset();
      },
      error: (error) => {
        let errorMessage = 'Error desconocido. Por favor, intenta de nuevo.';
        console.log(error);

        if (error && error.error && error.error.message) {
          errorMessage = error.error.message;
        } else if (error && error.message) {
          errorMessage = error.message;
        }

        this.modalService.showError('Error', errorMessage);
        this.errorMessage = 'Error al enviar la comunicación. Intenta de nuevo.';
      },
    });
  }
}
