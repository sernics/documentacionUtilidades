import { Component, inject, OnInit, ViewChild } from '@angular/core';
import { MatSidenav, MatSidenavContainer } from '@angular/material/sidenav';
import { MatSidenavModule } from '@angular/material/sidenav';
import { MatNavList } from '@angular/material/list';
import { RouterOutlet } from '@angular/router';
import { MatToolbarModule } from '@angular/material/toolbar';
import { AuthService } from '@app/services';
import { Router } from '@angular/router';
import { MatIconModule } from '@angular/material/icon';
import { FooterComponent } from '../footer/footer.component';
import { UserInfoService } from '@app/services/user-info/user-info.service';
import { MainMenuService } from '@app/services/mainMenuService/main-menu.service';

@Component({
  selector: 'app-layout',
  standalone: true,
  imports: [
    MatSidenavContainer,
    MatSidenavModule,
    MatNavList,
    RouterOutlet,
    MatToolbarModule,
    MatIconModule,
    FooterComponent,
  ],
  templateUrl: './layout.component.html',
  styleUrl: './layout.component.scss',
})
export class LayoutComponent implements OnInit {
  protected authService: AuthService = inject(AuthService);
  private router = inject(Router);
  private userInfo = inject(UserInfoService);
  private mainMenuService = inject(MainMenuService);
  @ViewChild('sidenav') sidenav!: MatSidenav;

  ngOnInit(): void {
    this.userInfo.setUserInfo();
  }

  /**
   * Toggles the sidenav and stops event propagation to prevent content click handler.
   * @param event The click event.
   */
  toggleSidenav(event: MouseEvent): void {
    event.stopPropagation(); // Prevent click from reaching mat-sidenav-content
    this.sidenav.toggle();
  }

  /**
   * Handles clicks on the content area to close the sidenav.
   * Ignores clicks on the toggle button.
   * @param event The click event.
   */
  onContentClick(event: MouseEvent): void {
    const target = event.target as HTMLElement;
    // Ignore clicks on the toggle button
    if (target.closest('button[mat-icon-button]') || target.closest('mat-sidenav')) {
      return;
    }
    this.sidenav.close();
  }

  /**
   * Navigates to the specified route if it is different from the current route.
   * If navigation occurs, it also closes the sidenav.
   *
   * @param route - The target route to navigate to.
   */
  go(route: string) {
    if (this.router.url !== route) {
      this.router.navigate([route]);
      this.sidenav.close();
    }
  }

  /**
   * Retrieves the main menu data from the `mainMenuService`.
   *
   * @returns The menu data provided by the `mainMenuService`.
   */
  getMenu() {
    return this.mainMenuService.getMenu();
  }

  /**
   * Logs the user out of the application by invoking the authentication service's logout method.
   * After logging out, navigates the user to the login page.
   */
  logout() {
    this.authService.logout();
    this.router.navigate(['/login']);
  }

  /**
   * Extracts and returns the first segment of a given title string, 
   * splitting it by the '/' character.
   *
   * @param title - The input string to process, expected to contain segments separated by '/'.
   * @returns The first segment of the input string before the '/' character.
   */
  getTitle(title: string) {
    return title.split('/')[0];
  }
}
