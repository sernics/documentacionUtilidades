import { Component, inject } from '@angular/core';
import { MatProgressSpinnerModule } from '@angular/material/progress-spinner';
import { Router } from '@angular/router';
import { PingService } from '@app/services/pingService/ping.service';

@Component({
  standalone: true,
  imports: [MatProgressSpinnerModule],
  templateUrl: './error.component.html',
  styleUrl: './error.component.scss'
})
export class ErrorComponent {
  private pingService = inject(PingService);
  private router = inject(Router);
  protected isLoading = false;

  /**
   * Retries the operation by invoking the ping service and handling the response.
   * - Sets `isLoading` to `true` before starting the operation.
   * - On a successful response, navigates to the root route (`/`) and sets `isLoading` to `false`.
   * - On an error response, sets `isLoading` to `false`.
   *
   * @remarks
   * This method uses the `pingService` to check connectivity or perform a health check.
   * It also utilizes Angular's `Router` to navigate upon success.
   */
  retry() {
    this.isLoading = true;
    this.pingService.ping().subscribe({
      next: () => {
        this.isLoading = false;
        this.router.navigate(['/']);
      },
      error: () => {
        this.isLoading = false;
      }
    })
  }
}

  <button mat-raised-button color="primary" (click)="retry()" [disabled]="isLoading">
