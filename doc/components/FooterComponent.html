import { Component } from '@angular/core';

@Component({
  selector: 'app-footer',
  imports: [],
  templateUrl: './footer.component.html',
  styleUrl: './footer.component.scss',
})
export class FooterComponent {
  mostrarAvisoLegal = false;
  mostrarTitularidadPortal = false;
  mostrarCondicionesUso = false;
  mostrarPropiedadIntelectual = false;
  mostrarPoliticaEnlaces = false;
  mostrarPoliticaSpam = false;
  mostrarExclusionResponsabilidad = false;
  mostrarJurisdiccion = false;

  /**
   * Toggles the visibility of the legal notice.
   *
   * This method switches the value of `mostrarAvisoLegal` between `true` and `false`.
   */
  toggleAvisoLegal() {
    this.mostrarAvisoLegal = !this.mostrarAvisoLegal;
  }

  /**
   * Handles the closing of the legal notice modal.
   *
   * @param event - The event triggered by the user interaction.
   * @remarks
   * This method checks if the event target has the class 'modal-overlay'.
   * If it does, it sets `mostrarAvisoLegal` to `false`, effectively closing the modal.
   */
  cerrarAvisoLegal(event: Event) {
    const target = event.target as HTMLElement;
    if (target.classList.contains('modal-overlay')) {
      this.mostrarAvisoLegal = false;
    }
  }

  /**
   * Toggles the boolean value of a section property on the component.
   *
   * @param seccion - The name of the section property to toggle.
   */
  toggleSeccion(seccion: string) {
    if (Object.prototype.hasOwnProperty.call(this, seccion)) {
      (this as any)[seccion] = !(this as any)[seccion];
    }
  }
}
