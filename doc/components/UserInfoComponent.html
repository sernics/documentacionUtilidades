import { Component, inject, OnInit } from '@angular/core';
import { UserInfoService } from '@app/services/user-info/user-info.service';

@Component({
  selector: 'app-user-info',
  imports: [],
  templateUrl: './user-info.component.html',
  styleUrl: './user-info.component.scss',
})
export class UserInfoComponent {
  protected userInfo = inject(UserInfoService);

  /**
   * Retrieves the name of the user from the user information.
   *
   * @returns {string | undefined} The name of the user, or undefined if the user information is not available.
   */
  getName() {
    return this.userInfo.getUserInfo()?.name;
  }

  /**
   * Retrieves the surname (sn) from the user information.
   *
   * @returns {string | undefined} The surname of the user, or undefined if not available.
   */
  getSurname() {
    return this.userInfo.getUserInfo()?.sn;
  }

  /**
   * Retrieves the user's name from the userInfo object.
   * The userName is extracted from the email address by splitting at the '@' character
   * and taking the part before it.
   *
   * @returns {string | undefined} The user's name or undefined if userInfo is not available.
   */
  getUserName() {
    return this.userInfo.getUserInfo()?.userName.split('@')[0];
  }

  /**
   * Retrieves the employee number (nomina) from the user information.
   *
   * @returns {string | undefined} The employee number, or undefined if not available.
   */
  getEmployeeNumber() {
    return this.userInfo.getUserInfo()?.nomina;
  }

  /**
   * Retrieves the email address of the user from the user information.
   *
   * @returns {string | undefined} The email address of the user, or undefined if not available.
   */
  getEmail() {
    return this.userInfo.getUserInfo()?.mail;
  }
}
