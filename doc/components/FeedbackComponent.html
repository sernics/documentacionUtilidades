import { Component, OnInit, OnDestroy, inject, Input } from '@angular/core';
import { Router } from '@angular/router';
import {
  ModalService,
  ModalState,
} from '@app/services/modalService/modal.service';
import { Subscription } from 'rxjs';

@Component({
  selector: 'app-feedback',
  templateUrl: './feedback.component.html',
  styleUrls: ['./feedback.component.scss'],
})
export class FeedbackComponent implements OnInit, OnDestroy {
  @Input() redirect: string = '/formularios';
  protected title: string = '';
  protected message: string = '';
  protected errorDetails: string | null = null;
  protected isVisible: boolean = false;
  private modalSubscription?: Subscription;
  private router = inject(Router);

  constructor(private modalService: ModalService) {
    console.log('FeedbackComponent constructor called');
  }

  /**
   * Lifecycle hook that is called after the component's view has been fully initialized.
   * Subscribes to the `modalState$` observable from the `modalService` to listen for changes
   * in the modal's state. Updates the component's properties (`title`, `message`, `errorDetails`,
   * and `isVisible`) based on the received modal state.
   *
   * Logs the initialization of the component and the received modal state to the console.
   * Handles errors that may occur during the subscription process.
   *
   * @returns void
   */
  ngOnInit() {
    console.log('FeedbackComponent ngOnInit called');
    this.modalSubscription = this.modalService.modalState$.subscribe(
      (state: ModalState) => {
        console.log('Modal state received:', state);
        this.title = state.title;
        this.message = state.message;
        this.errorDetails = state.errorDetails || null;
        this.isVisible = state.isVisible;
      },
      (error) => {
        console.error('Error in modal subscription:', error);
      }
    );
  }

  /**
   * Lifecycle hook that is called when the component is destroyed.
   * Cleans up resources by unsubscribing from the `modalSubscription` if it exists.
   * This helps to prevent memory leaks by ensuring that subscriptions are properly disposed of.
   */
  ngOnDestroy() {
    console.log('FeedbackComponent ngOnDestroy called');
    if (this.modalSubscription) {
      this.modalSubscription.unsubscribe();
    }
  }

  /**
   * Closes the modal and navigates to the specified redirect route.
   * After navigation, the page is reloaded to ensure the latest state.
   *
   * @remarks
   * This method uses `modalService.hide()` to close the modal,
   * and `router.navigate()` to perform the redirection. The `replaceUrl`
   * option is used to replace the current history entry, preventing
   * navigation back to the modal. Finally, `window.location.reload()`
   * is called to reload the page.
   */
  close() {
    this.modalService.hide();
    this.router.navigate([this.redirect], { replaceUrl: true }).then(() => {
      window.location.reload();
    });
  }

  /**
   * Handles the click event on the modal overlay.
   * If the click event's target is the overlay itself (not any child elements),
   * the modal is hidden, the user is redirected to the specified route, and the page is reloaded.
   *
   * @param event - The mouse event triggered by the click on the overlay.
   */
  onOverlayClick(event: MouseEvent) {
    if (event.target === event.currentTarget) {
      this.modalService.hide();
      this.router.navigate([this.redirect], { replaceUrl: true }).then(() => {
        window.location.reload();
      });
    }
  }
}

@if (isVisible) {
  <div class="feedback-overlay" (click)="onOverlayClick($event)">
    <div class="feedback-container" (click)="$event.stopPropagation()">
      <div class="feedback-header">
        <h2 class="title">{{ title }}</h2>
        <button class="close-btn-header" (click)="close()">×</button>
      </div>
      <div class="feedback-body">
        <p class="message">{{ message }}</p>
        @if (errorDetails) {
          <div class="error-details">
            <pre class="error-text">{{ errorDetails }}</pre>
          </div>
        }
      </div>
      <div class="feedback-footer"></div>
    </div>
  </div>
}
