import { Component, Input } from '@angular/core';
import { IResponseFicha } from '@app/models/responseSolicitud/iResponseFicha';
import { tResponse } from '@app/models/responseSolicitud/tResponse';

@Component({
  selector: 'app-ficha-plan-emergencia',
  imports: [],
  templateUrl: './ficha-plan-emergencia.component.html',
  styleUrl: './ficha-plan-emergencia.component.scss'
})
export class FichaPlanEmergenciaComponent {
  @Input() datos: tResponse | null = null;

  /**
   * Retrieves the name of the recipient (destinatario) from the `datos` property.
   * 
   * @returns {string | null} The recipient's name if available, or `null` if not.
   */
  getReceivers(): string | null {
    return this.datos !== null && (this.datos as IResponseFicha).nombreDestinatario !== null
      ? (this.datos as IResponseFicha).nombreDestinatario
      : null;
  }
}
