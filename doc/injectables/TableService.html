import { inject, Injectable, signal } from '@angular/core';
import { iEstadoForm } from '@app/models/iEstadoForm';
import { LocalStorageService } from '../localStorageService/local-storage-service.service';
import { localStorageEnum } from '@app/enums/localStorageEnum';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { iCabecerasList } from '@app/models/iCabecerasList';
import { EnviromentService } from '../environment/environment.service';

@Injectable({
  providedIn: 'root',
})
export class TableService {
  private localStorage = inject(LocalStorageService);
  private http = inject(HttpClient);
  private environment = inject(EnviromentService);
  private dataSignal = signal<iCabecerasList | null>(null);

  /**
   * Fetches and updates the data table based on the provided state information and page number.
   * Sends a PUT request to retrieve a list of solicitantes (requests) filtered by the given state.
   *
   * @param estadoInfo - An object containing the state information for filtering the data.
   * @param page - The page number to fetch.
   *
   * @remarks
   * - The method retrieves the authentication token from local storage and includes it in the request headers.
   * - On a successful response, the data is set using the `dataSignal` service.
   * - In case of an error, it logs the error to the console and sets the data to `null`.
   *
   * @throws Will log an error to the console if the HTTP request fails.
   */
  dataTable(estadoInfo: iEstadoForm, page: number) {
    estadoInfo.page = page;
    const token = this.localStorage.getItem(localStorageEnum.TOKEN) as string;
    const headers = new HttpHeaders({
      'Content-Type': 'application/json',
      'X-Auth-Token': token,
    });

    this.http
      .put<iCabecerasList>(
        `${this.environment.environmentConfig.API_URL}/formularios/getSolicitud/todos/estado`,
        { ...estadoInfo },
        { headers }
      )
      .subscribe({
        next: (data) => {
          this.dataSignal.set(data);
        },
        error: (err) => {
          console.error('Error fetching solicitantes TCP:', err);
          this.dataSignal.set(null);
        },
      });
  }

  /**
   * Retrieves the current data signal.
   *
   * @returns The data signal instance.
   */
  getDataSignal() {
    return this.dataSignal;
  }
}
