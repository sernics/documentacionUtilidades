import { Injectable, inject } from '@angular/core';
import { LocalStorageService } from '@app/services/index';
import { UserInfoService } from '@app/services/user-info/user-info.service';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { EnviromentService } from '../environment/environment.service';
import { Observable, catchError, map, of, switchMap } from 'rxjs';
import { iAuthenticate } from '@app/models/iAuthenticate';
import { RoleService } from '../roleService/role.service';
import { iUserMe } from '@app/models/iUserMe';
import { localStorageEnum } from '@app/enums/localStorageEnum';
import { Router } from '@angular/router';

@Injectable({
  providedIn: 'root',
})
export class AuthService {
  private localStorage = inject(LocalStorageService);
  private userInfo = inject(UserInfoService);
  private http = inject(HttpClient);
  private env = inject(EnviromentService);
  private roles = inject(RoleService);
  private router = inject(Router);
  protected logged: boolean = false;

  /**
   * Authenticates the user by sending their credentials to the server and returns an observable of the authentication response.
   *
   * @param username - The username of the user.
   * @param password - The password of the user, which will be base64 encoded before sending.
   * @returns An `Observable` of the authentication response.
   */
  getToken(username: string, password: string): Observable<iAuthenticate> {
    let formData = new FormData();
    formData.append('username', username);
    formData.append('password', btoa(password));
    return this.http
      .post<iAuthenticate>(
        `${this.env.environmentConfig.API_URL}/authenticate`,
        formData
      )
      .pipe();
  }

  /**
   * Logs in a user with the provided username and password.
   *
   * @param username - The username of the user.
   * @param password - The password of the user.
   * @param skip - Optional flag to skip the actual login process and use a default token. Defaults to false.
   * @returns An Observable that emits a boolean indicating the success or failure of the login process.
   *
   * @throws Will throw an error if the token information is undefined.
   */
  login(username: string, password: string, skip = false): Observable<boolean> {
    this.localStorage.setItem(localStorageEnum.USER, username);
    // TODO: eliminar esto
    if (skip) {
      this.localStorage.setItem(localStorageEnum.TOKEN, '1234567890');
      return of(true);
    }
    return this.getToken(username, password).pipe(
      switchMap((tokenInfo) => {
        if (!tokenInfo) {
          throw new Error('Token information is undefined');
        }
        this.localStorage.setItem(localStorageEnum.TOKEN, tokenInfo.token);
        const headers = new HttpHeaders({
          'Content-Type': 'application/json',
          'X-Auth-Token': tokenInfo.token,
        });
        return this.http
          .get<iUserMe>(`${this.env.environmentConfig.API_URL}/me`, {
            headers,
          })
          .pipe(
            map((userMe: iUserMe) => {
              this.roles.setRoles(userMe.roles);
            }),
            catchError((err) => {
              console.error('Error fetching user info', err);
              return of(false);
            })
          );
      }),
      map(() => {
        return true;
      }),
      catchError((err) => {
        console.error('Error during login', err);
        return of(false);
      })
    );
  }

  /**
   * Retrieves the name of the user from the user information.
   *
   * @returns {string | undefined} The name of the user, or undefined if the user information is not available.
   */
  getUser() {
    return this.userInfo.getUserInfo().name;
  }

  /**
   * Checks if the user is logged in by verifying the presence of 'user' and 'token' in local storage.
   *
   * @returns {boolean} - Returns `true` if both 'user' and 'token' are found in local storage, otherwise `false`.
   *
   * @remarks
   * This method checks the local storage for the existence of 'user' and 'token' items to determine if the user is logged in.
   * If either 'user' or 'token' is not found, it returns `false`.
   *
   * @todo
   * Implement authentication with the database if the user is not logged in.
   */
  isLoggedIn(): boolean {
    // Devuelve true si encuentra 'user' en las cookies
    // return this.localStorage.check('user') && this.localStorage.check('token');
    return (
      this.localStorage.getItem('user') !== undefined &&
      this.localStorage.getItem('token') !== undefined
    );
    // TODO Si no est√° hacer auth a la bd
  }

  /**
   * Checks if the user is currently logged in.
   *
   * @returns {boolean} - Returns `true` if the user is logged in, otherwise `false`.
   */
  loggedIn() {
    return this.logged;
  }

  /**
   * Logs out the current user by removing data from local storage.
   *
   * @remarks
   * This method currently only removes the user data from local storage.
   * A TODO is present to implement the logout functionality in the database.
   */
  logout() {
    const token = this.localStorage.getItem(localStorageEnum.TOKEN);
    this.localStorage.removeItem(localStorageEnum.USER);
    this.localStorage.removeItem(localStorageEnum.TOKEN);
    this.localStorage.removeItem(localStorageEnum.USER_ROLE);
    this.localStorage.removeItem(localStorageEnum.TIPO_SOLICITUD);
    const headers = new HttpHeaders({
      'Content-Type': 'application/json',
      'X-Auth-Token': token as string,
    });
    this.http.post(`${this.env.environmentConfig.API_URL}/logout`, {}, { headers }).subscribe({
      next: (response) => {
        console.log('Logout successful:', response);
      },
      error: (error) => {
        console.error('Logout error:', error);
      }
    });
  }
}
