import { inject, Injectable } from '@angular/core';
import { EnviromentService } from '../environment/environment.service';
import { HttpClient } from '@angular/common/http';
import { Router } from '@angular/router';
import { catchError, Observable, throwError, timeout } from 'rxjs';

@Injectable({
  providedIn: 'root'
})
export class PingService {
  private environment = inject(EnviromentService);
  private http = inject(HttpClient);
  private router = inject(Router);

  /**
   * Sends a ping request to the server to check connectivity.
   *
   * @returns An `Observable` that emits the server's response or an error if the request fails.
   * 
   * @remarks
   * - The request will timeout if no response is received within 2000 milliseconds.
   * - If the request fails, the user is redirected to the `/error` route.
   * - The error is rethrown as a new `Error` with the message 'Ping failed'.
   */
  ping(): Observable<any> {
    return this.http.get(`${this.environment.environmentConfig.API_URL}/ping`).pipe(
      timeout(2000),
      catchError((error) => {
        this.router.navigate(['/error']);
        return throwError(() => new Error('Ping failed'));
      })
    )
  }
}
